// --- JSON एक्सपोर्ट करने का फंक्शन --- window.exportJSON = function() { const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(transportData, null, 2)); const downloadAnchorNode = document.createElement('a'); downloadAnchorNode.setAttribute("href", dataStr); downloadAnchorNode.setAttribute("download", "transport_data.json"); document.body.appendChild(downloadAnchorNode); downloadAnchorNode.click(); downloadAnchorNode.remove(); } // --- अपडेटेड listenAndFill फंक्शन --- window.listenAndFill = function(fieldType, inputElement) { return new Promise((resolve, reject) => { let SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("Voice typing is not supported."); reject("Not supported"); return; } let recognition = new SpeechRecognition(); recognition.lang = 'en-IN'; recognition.interimResults = false; recognition.maxAlternatives = 1; // पहले से मौजूद टेक्स्ट को सहेजें (Append logic) let existingText = inputElement.value; inputElement.placeholder = "🎤 Listening..."; inputElement.classList.add('mic-active'); recognition.start(); recognition.onresult = function(event) { let spokenText = event.results[0][0].transcript; // पुरानी वैल्यू के आगे नई वैल्यू जोड़ें let combinedText = existingText + (existingText ? ", " : "") + spokenText; inputElement.value = processInputData(combinedText, fieldType); resolve(true); }; recognition.onend = function() { inputElement.classList.remove('mic-active'); }; }); }
